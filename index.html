1. Find the Smaller Angle

    var Minimal_Angle = (h, m) => {
    hrsAngle=(h*360)/12+(m*360)/(12*60)
    minAngle=(m*360)/60;
    angle=Math.abs(hrsAngle-minAngle)
    if(angle>180){
      angle=360-angle;
    }
    return angle
    };

    var Check_Leap = (year) => {

    {

4. Reverse a Number

    var Reverse_Number = (N) => {
    let rev =0;
    while(N>0){
        let r = N%10;
        rev=rev*10+r;
        N=parseInt(N/10);
    }
    return rev;
    };

    var Substring_Check = (S1, S2) => {
